# rpg-mcp Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build an MCP server that exposes rpg-encoder's repository analysis capabilities to LLM clients via stdio, with a background file watcher for live incremental updates.

**Architecture:** New `rpg-mcp` crate using the `rmcp` SDK (official Rust MCP SDK) with `#[tool_router]` macros. Single repo per instance. File watcher (`notify` crate) with 2s debounce runs incremental updates via `RpgEvolution`. Graph stored in memory behind `Arc<RwLock>`, persisted via `RpgStore`.

**Tech Stack:** rmcp 0.6 (MCP SDK), notify 8 (file watcher), rpg-encoder (existing), tokio, serde, schemars

**Spec:** `docs/superpowers/specs/2026-04-26-rpg-mcp-design.md`

```rust
use std::path::{Path, PathBuf};
use std::sync::Arc;
use tokio::sync::RwLock;
use rpg_encoder::{RpgGraph, RpgSnapshot, RpgStore};

#[derive(Debug, Clone)]
pub struct McpConfig {
    pub workspace: PathBuf,
    pub data_dir: PathBuf,
    pub hash_mode: HashMode,
    pub semantic: bool,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum HashMode {
    Mtime,
    Content,
}

impl McpConfig {
    pub fn from_env() -> anyhow::Result<Self> {
        let workspace = std::env::var("RPG_WORKSPACE")
            .map_err(|_| anyhow::anyhow!("RPG_WORKSPACE env var is required"))?;
        let workspace = PathBuf::from(&workspace);

        if !workspace.exists() {
            anyhow::bail!("RPG_WORKSPACE does not exist: {}", workspace.display());
        }

        let data_dir = std::env::var("RPG_DATA_DIR")
            .map(PathBuf::from)
            .unwrap_or_else(|_| workspace.join(".rpg-data"));

        let hash_mode = match std::env::var("RPG_HASH_MODE").as_deref() {
            Ok("content") => HashMode::Content,
            _ => HashMode::Mtime,
        };

        let semantic = std::env::var("RPG_SEMANTIC")
            .map(|v| v == "1" || v.to_lowercase() == "true")
            .unwrap_or(false);

        Ok(Self { workspace, data_dir, hash_mode, semantic })
    }
}

pub struct AppState {
    pub config: McpConfig,
    pub graph: Arc<RwLock<RpgGraph>>,
    pub snapshot: Arc<RwLock<RpgSnapshot>>,
    pub store: Arc<RwLock<Option<RpgStore>>>,
    pub parser_registry: Arc<rpg_encoder::ParserRegistry>,
}

impl AppState {
    pub fn new(
        config: McpConfig,
        graph: RpgGraph,
        snapshot: RpgSnapshot,
        store: Option<RpgStore>,
        registry: rpg_encoder::ParserRegistry,
    ) -> Self {
        Self {
            config,
            graph: Arc::new(RwLock::new(graph)),
            snapshot: Arc::new(RwLock::new(snapshot)),
            store: Arc::new(RwLock::new(store)),
            parser_registry: Arc::new(registry),
        }
    }

    pub async fn graph(&self) -> tokio::sync::RwLockReadGuard<'_, RpgGraph> {
        self.graph.read().await
    }
}

pub fn compute_dir_hash(dir: &Path, mode: HashMode) -> anyhow::Result<String> {
    let mut entries: Vec<(String, u64)> = Vec::new();
    for entry in walkdir::WalkDir::new(dir)
        .into_iter()
        .filter_entry(|e| {
            let name = e.file_name().to_string_lossy();
            name != ".git" && name != "target" && name != "node_modules"
                && !name.starts_with('.')
        })
        .filter_map(|e| e.ok())
    {
        let path = entry.path();
        if !path.is_file() {
            continue;
        }
        let relative = path.strip_prefix(dir).unwrap_or(path).to_string_lossy().to_string();
        let val = match mode {
            HashMode::Mtime => {
                std::fs::metadata(path)
                    .and_then(|m| m.modified())
                    .map(|t| t.duration_since(std::time::UNIX_EPOCH).unwrap_or_default().as_secs())
                    .unwrap_or(0)
            }
            HashMode::Content => {
                use std::io::Read;
                let mut hasher = sha2::Sha256::new();
                let mut file = match std::fs::File::open(path) {
                    Ok(f) => f,
                    Err(_) => continue,
                };
                let mut buf = [0u8; 8192];
                loop {
                    let n = match file.read(&mut buf) {
                        Ok(0) => break,
                        Ok(n) => n,
                        Err(_) => break,
                    };
                    hasher.update(&buf[..n]);
                }
                let hash = hasher.finalize();
                u64::from_le_bytes(hash[0..8].try_into().unwrap_or([0; 8]))
            }
        };
        entries.push((relative, val));
    }
    entries.sort_by(|a, b| a.0.cmp(&b.0));
    let mut hasher = sha2::Sha256::new();
    for (path, val) in &entries {
        hasher.update(path.as_bytes());
        hasher.update(&val.to_le_bytes());
    }
    Ok(hex::encode(hasher.finalize()))
}

pub fn load_dir_hash(data_dir: &Path) -> Option<String> {
    let hash_path = data_dir.join(".rpg").join("dir_hash");
    std::fs::read_to_string(hash_path).ok().map(|s| s.trim().to_string())
}

pub fn save_dir_hash(data_dir: &Path, hash: &str) -> anyhow::Result<()> {
    let hash_path = data_dir.join(".rpg").join("dir_hash");
    if let Some(parent) = hash_path.parent() {
        std::fs::create_dir_all(parent)?;
    }
    std::fs::write(hash_path, hash)?;
    Ok(())
}
```

```rust
use std::path::PathBuf;
use std::sync::Arc;
use std::time::Duration;

use notify::{Event, EventKind, RecursiveMode, Watcher, recommended_watcher};
use tokio::sync::{mpsc, RwLock};

use crate::state::{AppState, compute_dir_hash, save_dir_hash};

const DEBOUNCE_MS: u64 = 2000;

pub struct FileWatcher {
    _watcher: Option<notify::RecommendedWatcher>,
    shutdown_tx: Option<mpsc::Sender<()>>,
}

impl Drop for FileWatcher {
    fn drop(&mut self) {
        if let Some(tx) = self.shutdown_tx.take() {
            let _ = tx.try_send(());
        }
    }
}

impl FileWatcher {
    pub fn start(state: Arc<AppState>) -> anyhow::Result<Self> {
        let (file_tx, mut file_rx) = mpsc::channel::<Vec<PathBuf>>(100);
        let (shutdown_tx, mut shutdown_rx) = mpsc::channel::<()>(1);

        let workspace = state.config.workspace.clone();
        let data_dir = state.config.data_dir.clone();
        let hash_mode = state.config.hash_mode;

        let mut watcher = recommended_watcher(move |res: Result<Event, notify::Error>| {
            if let Ok(event) = res {
                if event.kind.is_create() || event.kind.is_modify() || event.kind.is_remove() {
                    let paths: Vec<PathBuf> = event.paths.clone();
                    let _ = file_tx.blocking_send(paths);
                }
            }
        })?;

        watcher.watch(&workspace, RecursiveMode::Recursive)?;

        let shutdown_tx_clone = shutdown_tx.clone();

        tokio::spawn(async move {
            let mut pending: Vec<PathBuf> = Vec::new();
            let mut debounce = tokio::time::interval(Duration::from_millis(DEBOUNCE_MS));
            debounce.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Delay);

            loop {
                debounce.reset();
                tokio::select! {
                    Some(paths) = file_rx.recv() => {
                        for p in paths {
                            if !pending.contains(&p) {
                                pending.push(p);
                            }
                        }
                    }
                    _ = shutdown_rx.recv() => {
                        tracing::info!("File watcher shutting down");
                        break;
                    }
                    _ = debounce.tick() => {
                        if !pending.is_empty() {
                            let batch = std::mem::take(&mut pending);
                            if let Err(e) = process_changes(&state, &batch).await {
                                tracing::error!("Failed to process file changes: {}", e);
                            }
                        }
                    }
                }
            }

            let _ = shutdown_tx_clone;
        });

        Ok(Self {
            _watcher: Some(watcher),
            shutdown_tx: Some(shutdown_tx),
        })
    }
}

async fn process_changes(state: &AppState, changed_files: &[PathBuf]) -> anyhow::Result<()> {
    let workspace = &state.config.workspace;
    let data_dir = &state.config.data_dir;
    let hash_mode = state.config.hash_mode;

    tracing::info!("Processing {} changed file(s)", changed_files.len());

    let snapshot = state.snapshot.read().await;
    let diff = rpg_encoder::generate_diff(&snapshot, workspace, &state.parser_registry)?;
    drop(snapshot);

    if diff.is_empty() {
        tracing::debug!("No meaningful changes detected");
        return Ok(());
    }

    let mut snapshot = state.snapshot.write().await;
    let mut evolution = rpg_encoder::RpgEvolution::new(&mut snapshot, &state.parser_registry);
    let summary = evolution.process_diff(diff, None).await?;
    drop(evolution);

    let graph = snapshot.graph.clone();
    drop(snapshot);

    *state.graph.write().await = graph;

    if let Some(store) = state.store.write().await.as_mut() {
        let patch = rpg_encoder::Patch::from_evolution_summary(
            &summary,
            std::time::SystemTime::now()
                .duration_since(std::time::UNIX_EPOCH)
                .unwrap_or_default()
                .as_secs(),
        );
        store.write_patch(&patch)?;
    }

    let new_hash = compute_dir_hash(workspace, hash_mode)?;
    save_dir_hash(data_dir, &new_hash)?;

    tracing::info!(
        "Graph updated: +{} files, ~{} files, -{} files ({} nodes, {} edges)",
        summary.files_added, summary.files_modified, summary.files_deleted,
        state.graph.read().await.node_count(),
        state.graph.read().await.edge_count()
    );

    Ok(())
}
```

**Note:** `Patch::from_evolution_summary` may not exist in the current API. The implementer should check `rpg-encoder/src/storage/patch.rs` for the actual `Patch` constructor and adapt accordingly. If `Patch` cannot be constructed from `EvolutionSummary`, fall back to saving a new base snapshot via `store.save_base(&snapshot)`.

```rust
use std::sync::Arc;

use rmcp::{
    ErrorData as McpError, RoleServer, ServerHandler,
    handler::server::router::tool::ToolRouter,
    model::*,
    tool, tool_handler, tool_router,
    schemars,
    service::RequestContext,
};
use serde::{Deserialize, Serialize};

use crate::state::AppState;

#[derive(Debug, Deserialize, schemars::JsonSchema)]
pub struct SearchNodesParams {
    pub query: String,
    pub kind: Option<String>,
    pub category: Option<String>,
    pub limit: Option<usize>,
}

#[derive(Debug, Deserialize, schemars::JsonSchema)]
pub struct GetNodeParams {
    pub node_id: String,
}

#[derive(Debug, Deserialize, schemars::JsonSchema)]
pub struct GetEdgesParams {
    pub source_id: Option<String>,
    pub target_id: Option<String>,
    pub edge_type: Option<String>,
    pub limit: Option<usize>,
}

#[derive(Debug, Deserialize, schemars::JsonSchema)]
pub struct GetFeaturesParams {
    pub file_path: Option<String>,
}

#[derive(Clone)]
pub struct RpgService {
    state: Arc<AppState>,
    tool_router: ToolRouter<RpgService>,
}

#[tool_router]
impl RpgService {
    pub fn new(state: Arc<AppState>) -> Self {
        Self {
            state,
            tool_router: Self::tool_router(),
        }
    }

    #[tool(description = "Encode the workspace into an RPG graph (full re-encode). Replaces current graph.")]
    async fn encode_repo(&self) -> Result<CallToolResult, McpError> {
        let start = std::time::Instant::now();
        let workspace = self.state.config.workspace.clone();
        let data_dir = self.state.config.data_dir.clone();
        let hash_mode = self.state.config.hash_mode;

        let mut encoder = rpg_encoder::RpgEncoder::new()
            .map_err(|e| McpError::internal_error(format!("Failed to create encoder: {}", e), None))?;

        let result = encoder.encode(&workspace)
            .map_err(|e| McpError::internal_error(format!("Encode failed: {}", e), None))?;

        let graph = result.graph.clone();

        if let Err(e) = std::fs::create_dir_all(&data_dir) {
            tracing::warn!("Failed to create data dir: {}", e);
        }

        let snapshot = rpg_encoder::RpgSnapshot::from_encoder(&encoder);

        let mut store_lock = self.state.store.write().await;
        let store = match rpg_encoder::RpgStore::init(&data_dir) {
            Ok(s) => {
                store_lock.replace(s);
                store_lock.as_mut().unwrap()
            }
            Err(_) => match rpg_encoder::RpgStore::open(&data_dir) {
                Ok(s) => {
                    store_lock.replace(s);
                    store_lock.as_mut().unwrap()
                }
                Err(e) => {
                    tracing::error!("Failed to open store: {}", e);
                    return Err(McpError::internal_error(format!("Store error: {}", e), None));
                }
            },
        };

        store.save_base(&snapshot)
            .map_err(|e| McpError::internal_error(format!("Failed to save: {}", e), None))?;
        drop(store_lock);

        if let Ok(hash) = crate::state::compute_dir_hash(&workspace, hash_mode) {
            let _ = crate::state::save_dir_hash(&data_dir, &hash);
        }

        *self.state.graph.write().await = graph.clone();
        *self.state.snapshot.write().await = snapshot;

        let output = serde_json::json!({
            "files_processed": result.files_processed,
            "files_skipped": result.files_skipped,
            "parse_errors": result.parse_errors.len(),
            "nodes": graph.node_count(),
            "edges": graph.edge_count(),
            "duration_ms": start.elapsed().as_millis()
        });

        Ok(CallToolResult::success(vec![Content::text(output.to_string())]))
    }

    #[tool(description = "Get a summary of the current graph: node/edge counts, languages, edge types")]
    async fn get_graph_summary(&self) -> Result<CallToolResult, McpError> {
        let graph = self.state.graph.read().await;
        let report = rpg_encoder::encoder::ValidationReport::from_graph(&graph);

        let mut languages: Vec<String> = graph.nodes()
            .map(|n| n.language.clone())
            .filter(|l| !l.is_empty())
            .collect::<std::collections::HashSet<_>>()
            .into_iter()
            .collect();
        languages.sort();

        let output = serde_json::json!({
            "total_nodes": report.total_nodes,
            "total_edges": report.total_edges,
            "languages": languages,
            "edge_types": report.edge_type_counts,
            "node_categories": report.node_category_counts,
            "import_resolution_rate": format!("{:.1}%", report.import_resolution_rate * 100.0),
            "calls": report.call_edge_count,
            "implements": report.implements_edge_count,
            "ffi": report.ffi_edge_count,
        });

        Ok(CallToolResult::success(vec![Content::text(output.to_string())]))
    }

    #[tool(description = "Search graph nodes by name substring, with optional kind/category filters")]
    async fn search_nodes(&self, params: SearchNodesParams) -> Result<CallToolResult, McpError> {
        let graph = self.state.graph.read().await;
        let limit = params.limit.unwrap_or(50);
        let query = params.query.to_lowercase();

        let mut results = Vec::new();
        for node in graph.nodes() {
            if results.len() >= limit {
                break;
            }
            if !node.name.to_lowercase().contains(&query) {
                continue;
            }
            if let Some(ref kind) = params.kind {
                if node.kind != *kind {
                    continue;
                }
            }
            if let Some(ref cat) = params.category {
                if format!("{:?}", node.category).to_lowercase() != cat.to_lowercase() {
                    continue;
                }
            }

            results.push(serde_json::json!({
                "id": format!("node_{}", node.id.index()),
                "name": node.name,
                "kind": node.kind,
                "category": format!("{:?}", node.category).to_lowercase(),
                "language": node.language,
                "path": node.path.as_ref().map(|p| p.to_string_lossy().to_string()),
                "signature": node.signature,
                "description": node.description,
            }));
        }

        Ok(CallToolResult::success(vec![Content::text(serde_json::to_string_pretty(&results).unwrap_or_default())]))
    }

    #[tool(description = "Get full details of a specific node by ID, including connected edges")]
    async fn get_node_details(&self, params: GetNodeParams) -> Result<CallToolResult, McpError> {
        let graph = self.state.graph.read().await;
        let node_id_str = params.node_id.trim_start_matches("node_");
        let idx: usize = node_id_str.parse()
            .map_err(|_| McpError::invalid_params("node_id must be 'node_N' where N is a number", None))?;

        let node = graph.get_node(rpg_encoder::NodeId::new(idx))
            .ok_or_else(|| McpError::invalid_params(&format!("Node {} not found", params.node_id), None))?;

        let mut incoming = Vec::new();
        let mut outgoing = Vec::new();
        for (src, tgt, edge) in graph.edges() {
            if tgt == node.id {
                if let Some(src_node) = graph.get_node(src) {
                    incoming.push(serde_json::json!({
                        "source": src_node.name,
                        "source_id": format!("node_{}", src.index()),
                        "type": format!("{:?}", edge.edge_type).to_lowercase(),
                    }));
                }
            }
            if src == node.id {
                if let Some(tgt_node) = graph.get_node(tgt) {
                    outgoing.push(serde_json::json!({
                        "target": tgt_node.name,
                        "target_id": format!("node_{}", tgt.index()),
                        "type": format!("{:?}", edge.edge_type).to_lowercase(),
                    }));
                }
            }
        }

        let output = serde_json::json!({
            "id": format!("node_{}", node.id.index()),
            "name": node.name,
            "kind": node.kind,
            "category": format!("{:?}", node.category).to_lowercase(),
            "language": node.language,
            "path": node.path.as_ref().map(|p| p.to_string_lossy().to_string()),
            "signature": node.signature,
            "description": node.description,
            "features": node.features,
            "feature_path": node.feature_path,
            "semantic_feature": node.semantic_feature,
            "documentation": node.documentation,
            "metadata": node.metadata,
            "incoming_edges": incoming,
            "outgoing_edges": outgoing,
        });

        Ok(CallToolResult::success(vec![Content::text(serde_json::to_string_pretty(&output).unwrap_or_default())]))
    }

    #[tool(description = "Query graph edges with optional source/target/type filters")]
    async fn get_edges(&self, params: GetEdgesParams) -> Result<CallToolResult, McpError> {
        let graph = self.state.graph.read().await;
        let limit = params.limit.unwrap_or(100);

        let mut results = Vec::new();
        for (src, tgt, edge) in graph.edges() {
            if results.len() >= limit {
                break;
            }
            if let Some(ref src_id) = params.source_id {
                let idx_str = src_id.trim_start_matches("node_");
                if let Ok(idx) = idx_str.parse::<usize>() {
                    if src != rpg_encoder::NodeId::new(idx) {
                        continue;
                    }
                }
            }
            if let Some(ref tgt_id) = params.target_id {
                let idx_str = tgt_id.trim_start_matches("node_");
                if let Ok(idx) = idx_str.parse::<usize>() {
                    if tgt != rpg_encoder::NodeId::new(idx) {
                        continue;
                    }
                }
            }
            if let Some(ref et) = params.edge_type {
                if format!("{:?}", edge.edge_type).to_lowercase() != et.to_lowercase() {
                    continue;
                }
            }

            let src_name = graph.get_node(src).map(|n| n.name.clone()).unwrap_or_default();
            let tgt_name = graph.get_node(tgt).map(|n| n.name.clone()).unwrap_or_default();

            results.push(serde_json::json!({
                "source_id": format!("node_{}", src.index()),
                "source_name": src_name,
                "target_id": format!("node_{}", tgt.index()),
                "target_name": tgt_name,
                "edge_type": format!("{:?}", edge.edge_type).to_lowercase(),
            }));
        }

        Ok(CallToolResult::success(vec![Content::text(serde_json::to_string_pretty(&results).unwrap_or_default())]))
    }

    #[tool(description = "Get the repository skeleton — files, units, kinds, visibility")]
    async fn get_skeleton(&self) -> Result<CallToolResult, McpError> {
        let graph = self.state.graph.read().await;

        let mut files = Vec::new();
        for node in graph.nodes() {
            if node.category != rpg_encoder::NodeCategory::File {
                continue;
            }
            let path = node.path.as_ref()
                .map(|p| p.to_string_lossy().to_string())
                .unwrap_or_default();

            let children: Vec<serde_json::Value> = graph.edges()
                .filter(|(s, _, e)| *s == node.id && e.edge_type == rpg_encoder::EdgeType::Contains)
                .filter_map(|(_, t, _)| graph.get_node(t).map(|n| {
                    serde_json::json!({
                        "name": n.name,
                        "kind": n.kind,
                        "category": format!("{:?}", n.category).to_lowercase(),
                    })
                }))
                .collect();

            files.push(serde_json::json!({
                "path": path,
                "language": node.language,
                "children": children,
            }));
        }

        Ok(CallToolResult::success(vec![Content::text(serde_json::to_string_pretty(&files).unwrap_or_default())]))
    }

    #[tool(description = "Get feature tree — entities with semantic features, descriptions, paths")]
    async fn get_features(&self, params: GetFeaturesParams) -> Result<CallToolResult, McpError> {
        let graph = self.state.graph.read().await;

        let mut features = Vec::new();
        for node in graph.nodes() {
            if node.category == rpg_encoder::NodeCategory::File
                || node.category == rpg_encoder::NodeCategory::Directory
                || node.category == rpg_encoder::NodeCategory::Repository
            {
                continue;
            }
            if let Some(ref filter_path) = params.file_path {
                let node_path = node.path.as_ref()
                    .map(|p| p.to_string_lossy().to_string())
                    .unwrap_or_default();
                if !node_path.contains(filter_path) {
                    continue;
                }
            }
            if node.features.is_empty() && node.description.is_none() {
                continue;
            }

            features.push(serde_json::json!({
                "name": node.name,
                "kind": node.kind,
                "path": node.path.as_ref().map(|p| p.to_string_lossy().to_string()),
                "features": node.features,
                "description": node.description,
                "feature_path": node.feature_path,
                "semantic_feature": node.semantic_feature,
            }));
        }

        Ok(CallToolResult::success(vec![Content::text(serde_json::to_string_pretty(&features).unwrap_or_default())]))
    }

    #[tool(description = "Get architectural components and their relationships")]
    async fn get_components(&self) -> Result<CallToolResult, McpError> {
        let graph = self.state.graph.read().await;

        let mut components: Vec<serde_json::Value> = Vec::new();
        for node in graph.nodes() {
            if node.category != rpg_encoder::NodeCategory::Component {
                continue;
            }
            let children: Vec<serde_json::Value> = graph.edges()
                .filter(|(s, _, e)| *s == node.id && e.edge_type == rpg_encoder::EdgeType::Contains)
                .filter_map(|(_, t, _)| graph.get_node(t).map(|n| n.name.clone()))
                .collect();

            components.push(serde_json::json!({
                "name": node.name,
                "description": node.description,
                "features": node.features,
                "nodes": children,
            }));
        }

        if components.is_empty() {
            components.push(serde_json::json!({
                "message": "No components found. Components are created during semantic enrichment (RPG_SEMANTIC=true)."
            }));
        }

        Ok(CallToolResult::success(vec![Content::text(serde_json::to_string_pretty(&components).unwrap_or_default())]))
    }
}
```

```rust
use std::sync::Arc;

use rmcp::ServiceExt;
use service::RpgService;
use state::{AppState, McpConfig, compute_dir_hash, load_dir_hash, save_dir_hash};

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    let debug = std::env::var("RPG_DEBUG")
        .map(|v| v == "1" || v.to_lowercase() == "true")
        .unwrap_or(false);

    let filter = if debug {
        EnvFilter::from_default_env().add_directive("rpg_mcp=debug".parse()?)
    } else {
        EnvFilter::from_default_env().add_directive("rpg_mcp=warn".parse()?)
    };

    tracing_subscriber::fmt()
        .with_env_filter(filter)
        .with_writer(std::io::stderr)
        .with_ansi(false)
        .init();

    let config = McpConfig::from_env()?;
    tracing::info!("Workspace: {}", config.workspace.display());
    tracing::info!("Data dir: {}", config.data_dir.display());

    let mut encoder = rpg_encoder::RpgEncoder::new()?;
    let registry = encoder.parser_registry();

    let graph = load_or_encode(&config, &mut encoder).await?;
    let snapshot = rpg_encoder::RpgSnapshot::from_encoder(&encoder);

    if let Err(e) = std::fs::create_dir_all(&config.data_dir) {
        tracing::warn!("Failed to create data dir: {}", e);
    }

    let store = match rpg_encoder::RpgStore::open(&config.data_dir) {
        Ok(s) => Some(s),
        Err(_) => rpg_encoder::RpgStore::init(&config.data_dir).ok(),
    };

    let state = Arc::new(AppState::new(config.clone(), graph, snapshot, store, registry));
    let _watcher = watcher::FileWatcher::start(state.clone())?;

    let service = RpgService::new(state).serve(rmcp::transport::stdio()).await?;
    service.waiting().await?;

    Ok(())
}

async fn load_or_encode(
    config: &McpConfig,
    encoder: &mut rpg_encoder::RpgEncoder,
) -> anyhow::Result<rpg_encoder::RpgGraph> {
    let current_hash = compute_dir_hash(&config.workspace, config.hash_mode)?;
    let stored_hash = load_dir_hash(&config.data_dir);

    if let Some(ref stored) = stored_hash {
        if stored == &current_hash {
            tracing::info!("Hash matches, loading persisted graph");
            if let Ok(store) = rpg_encoder::RpgStore::open(&config.data_dir) {
                if let Ok(snapshot) = store.load() {
                    tracing::info!("Loaded {} nodes, {} edges",
                        snapshot.graph.node_count(),
                        snapshot.graph.edge_count());
                    return Ok(snapshot.graph);
                }
            }
            tracing::warn!("Persisted graph corrupt, re-encoding");
        } else {
            tracing::info!("Hash mismatch, re-encoding");
        }
    } else {
        tracing::info!("No persisted graph, encoding workspace");
    }

    let result = encoder.encode(&config.workspace)?;
    let hash = current_hash;
    save_dir_hash(&config.data_dir, &hash)?;
    tracing::info!("Encoded {} files: {} nodes, {} edges",
        result.files_processed, result.graph.node_count(), result.graph.edge_count());

    Ok(result.graph)
}
```
